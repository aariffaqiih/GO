// menghitung jumlah kuantitas bilangan genap! misalkan kuantitas = 5, maka jumlah bilangan = 2 + 4 + 6 + 8 + 10 = 30.

package main

import "fmt"
func main() {
	var n int; fmt.Scan(&n)
	var penjumlahan_hasil int
	for i := 1; i <= n; i++ {
		angka_sebanyak_n := 2 * i
		penjumlahan_hasil += angka_sebanyak_n
	}
	fmt.Print(penjumlahan_hasil)
}

// menghitung kuantitas bilangan positif dan negatif dari serangkaian bilangan yang diinputkan. program berakhir jika input 0.

package main

import "fmt"
func main() {
    var berhenti0 int; fmt.Scan(&berhenti0)
    kuantitasPositif := 0
    kuantitasNegatif := 0
    for berhenti0 != 0 {
        fmt.Scan(&berhenti0)
        if berhenti0 >= 0 {
            kuantitasPositif++
        } else if berhenti0 < 0 {
            kuantitasNegatif++
        }
    }
    fmt.Print(kuantitasPositif, kuantitasNegatif)
}

// menu

package main

import "fmt"
func main() {
	fmt.Println("1: penjumlahan")
	fmt.Println("2: perkalian")
	fmt.Println("3: cek prima")
	fmt.Println("4: keluar")
	var menu int
	for menu != 4 {
		fmt.Print("menu: "); fmt.Scan(&menu)
		switch menu {
		case 1:
			var a int; fmt.Print("a: "); fmt.Scan(&a)
			var b int; fmt.Print("b: "); fmt.Scan(&b)
			fmt.Println("a + b = ", a+b)
		case 2:
			var a int; fmt.Print("a: "); fmt.Scan(&a)
			var b int; fmt.Print("b: "); fmt.Scan(&b)
			fmt.Println("a x b = ", a*b)
		case 3:
			var c int; fmt.Print("c: "); fmt.Scan(&c)
			isPrime := true
			if c <= 1 {
				isPrime = false
			} else {
				i := 2
				for i < c && isPrime {
					if c%i == 0 {
						isPrime = false
					}
					i++
				}
			}
			if isPrime {
				fmt.Println(c, "prima")
			} else {
				fmt.Println(c, "tidak prima")
			}
		case 4:
			fmt.Println("program selesai")
		default:
			fmt.Println("hanya ada menu 1, 2, dan 4!")
		}
	}
}
